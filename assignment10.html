# Do a simulation example first -- breaks in a regression, no dynamics
set.seed(123)
y1 <- rnorm(50, mean=0, sd=1)     # Break at 50
y2 <- rnorm(100, mean=2, sd=0.5)  # Break at 150
y3 <- rnorm(50, mean=-1, sd=2)
y <- c(y1,y2,y3)
plot(ts(y))
